34 · Writing a Custom Scheduler

Standard section: [exec.sched] stdexec reference: exec/inline_scheduler.hpp, exec/trampoline_scheduler.hpp Related tests: tests/test_custom_scheduler.cpp

Overview

Writing a custom scheduler allows you to plug your own execution resources (thread pools, hardware accelerators, event loops) into the std::execution ecosystem.

Scheduler Requirements

cpp
template<class Sch>
concept scheduler =
  std::destructible<Sch> &&
  std::copy_constructible<Sch> &&
  std::equality_comparable<Sch> &&
  requires(Sch&& sch, const Sch& csch) {
    { schedule(static_cast<Sch&&>(sch)) } -> sender;
    { csch.query(get_forward_progress_guarantee) }
      -> same_as<forward_progress_guarantee>;
  };

Minimal Implementation: inline_scheduler

exec::inline_scheduler is the simplest scheduler reference implementation:

cpp
// Corresponding test: test_custom_scheduler.cpp → "inline_scheduler runs on current thread"
class inline_scheduler {
 public:
  // 1. Implement get_forward_progress_guarantee
  friend forward_progress_guarantee tag_invoke(
    tag_t<get_forward_progress_guarantee>,
    const inline_scheduler&) noexcept {
    return forward_progress_guarantee::concurrent;
  }

  // 2. Equality comparison
  bool operator==(const inline_scheduler&) const = default;
};

// 3. Define inline_scheduler's schedule sender
struct inline_schedule_sender {
  using sender_concept = sender_tag;

  // 4. Completion signatures
  using completion_signatures =
    completion_signatures<set_value_t()>;

  // 5. Operation state
  template <receiver R>
  struct op {
    R rcvr_;
    void start() & noexcept {
      // Complete immediately on the current thread
      set_value(static_cast<R&&>(rcvr_));
    }
  };

  // 6. connect
  template <receiver R>
  auto connect(R r) && -> op<R> {
    return op<R>{static_cast<R&&>(r)};
  }
};

// 7. Implement the schedule CPO via tag_invoke
inline_schedule_sender tag_invoke(
  tag_t<schedule>, const inline_scheduler&) {
  return {};
}

Advanced Implementation: trampoline_scheduler

trampoline_scheduler demonstrates a more complex pattern — scheduling execution within an event loop:

cpp
class trampoline_scheduler {
 public:
  friend forward_progress_guarantee tag_invoke(
    tag_t<get_forward_progress_guarantee>,
    const trampoline_scheduler&) noexcept {
    return forward_progress_guarantee::concurrent;
  }

  bool operator==(const trampoline_scheduler&) const = default;
};

// Trampoline sender does not propagate work immediately;
// instead it enqueues work to be executed one by one in the event loop
struct trampoline_schedule_sender {
  using sender_concept = sender_tag;
  using completion_signatures =
    completion_signatures<set_value_t()>;

  template <receiver R>
  struct op {
    R rcvr_;

    void start() & noexcept {
      // Enqueue work; it will be executed by the event loop after the current function returns
      trampoline_queue::enqueue([this] {
        set_value(static_cast<R&&>(rcvr_));
      });
    }
  };

  template <receiver R>
  auto connect(R r) && -> op<R> {
    return op<R>{static_cast<R&&>(r)};
  }
};

trampoline_schedule_sender tag_invoke(
  tag_t<schedule>, const trampoline_scheduler&) {
  return {};
}

Custom Thread Pool Scheduler

A complete thread pool scheduler implementation framework:

cpp
// Corresponding test: test_custom_scheduler.cpp → "static_thread_pool scheduler is a valid scheduler"
class thread_pool {
  // Work queue and thread management
  std::vector<std::thread> threads_;
  std::queue<std::function<void()>> queue_;
  std::mutex mtx_;
  std::condition_variable cv_;
  bool done_ = false;

public:
  class scheduler {
    thread_pool* pool_;
  public:
    explicit scheduler(thread_pool* pool) : pool_(pool) {}

    friend forward_progress_guarantee tag_invoke(
      tag_t<get_forward_progress_guarantee>,
      const scheduler&) noexcept {
      return forward_progress_guarantee::parallel;
    }

    bool operator==(const scheduler& other) const = default;
  };

  scheduler get_scheduler() { return scheduler{this}; }

  // schedule sender
  struct schedule_sender {
    thread_pool* pool_;

    using sender_concept = sender_tag;
    using completion_signatures =
      completion_signatures<set_value_t()>;

    template <receiver R>
    struct op {
      thread_pool* pool_;
      R rcvr_;

      void start() & noexcept {
        pool_->enqueue([this] {
          set_value(static_cast<R&&>(rcvr_));
        });
      }
    };

    template <receiver R>
    auto connect(R r) && -> op<R> {
      return op<R>{pool_, static_cast<R&&>(r)};
    }
  };

  friend schedule_sender tag_invoke(
    tag_t<schedule>, const scheduler& sch) {
    return schedule_sender{sch.pool_};
  }
};

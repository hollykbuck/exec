41 · Appendix: Porting libunifex Synchronization Primitives

Standard Section: — stdexec Reference: tests/test_async_mutex.cpp, tests/test_async_mutex_v2.cpp, tests/test_async_manual_reset_event.cpp, tests/test_async_pass.cpp libunifex Reference: include/unifex/v1/async_mutex.hpp, include/unifex/v2/async_mutex.hpp, include/unifex/v1/async_manual_reset_event.hpp, include/unifex/v2/async_manual_reset_event.hpp, include/unifex/async_pass.hpp

Overview

libunifex (Facebook/Meta) is the early reference implementation of P2300, containing several useful synchronization primitives. This appendix demonstrates how to port these primitives to stdexec, focusing on:

  • async_mutex — Asynchronous mutex, supports try_lock() / async_lock() / unlock()
  • async_mutex (v2) — v2 adds cancellation support (stop token) and Dekker synchronization pattern
  • async_manual_reset_event — Manual reset event, supports set() / reset() / async_wait()
  • async_pass — One-to-one synchronous channel (rendezvous), connecting caller and acceptor

The first three primitives share v1’s tagged pointer / queue architecture plus v2’s stop token cancellation framework.

Algorithm Flow

try_lock(): Single-step CAS — from INACTIVE(1) → 0. Lock acquired on success.

cpp
bool try_lock() noexcept {
  uintptr_t expected = INACTIVE;
  return state_.compare_exchange_strong(
    expected, 0, std::memory_order_acquire, std::memory_order_relaxed);
}

async_lock().start(): Calls try_enqueue(this):

  • If mutex is inactive → CAS to 0 (mark active), returns false → lock acquired synchronously, immediately set_value
  • If mutex is already active → push to stack, returns true → wait for unlock wakeup
cpp
void start() & noexcept {
  if (!mutex_.try_enqueue(this)) {
    // Lock acquired, synchronous completion
    ex::set_value(static_cast<R&&>(rcvr_));
  }
  // Otherwise already in wait queue
}

unlock(): Two-phase algorithm:

graph TD
  A["unlock()"]
  A --> B{"pending queue non-empty?"}
  B -->|YES| C["pop a waiter, resume it<br/>lock retained"]
  B -->|NO| D["try_mark_inactive_or_dequeue_all()"]
  D --> E{"Atomic queue empty?"}
  E -->|YES| F["Mark INACTIVE<br/>lock released"]
  E -->|NO| G["Dequeue all → reverse to FIFO<br/>push to pending queue → pop first, resume"]
  A --> H["Lock transferred to next waiter"]
cpp
void unlock() noexcept {
  if (pending_head_) {                       // 1) Prefer local pending queue
    waiter_base* w = pending_head_;
    pending_head_ = w->next_;
    w->resume_(w);
    return;
  }

  waiter_base* list = try_mark_inactive_or_dequeue_all();  // 2) Atomic operation
  if (!list) return;  // Marked INACTIVE, lock released

  // Reverse stack → FIFO pending queue
  // ... (reverse code)
  pending_head_ = reversed;

  // Pop first and resume (it holds the lock)
  waiter_base* w = pending_head_;
  pending_head_ = w->next_;
  w->resume_(w);
}

Complete Implementation (stdexec)

cpp
class async_mutex {
  struct waiter_base {
    waiter_base* next_ = nullptr;
    void (*resume_)(waiter_base*) noexcept;
  };

  static constexpr uintptr_t INACTIVE = 1;
  std::atomic<uintptr_t> state_{INACTIVE};

  waiter_base* pending_head_ = nullptr;
  waiter_base* pending_tail_ = nullptr;

  class lock_sender;

public:
  async_mutex() noexcept = default;

  bool try_lock() noexcept {
    uintptr_t expected = INACTIVE;
    return state_.compare_exchange_strong(
      expected, 0, std::memory_order_acquire, std::memory_order_relaxed);
  }

  void unlock() noexcept {
    if (pending_head_) {
      waiter_base* w = pending_head_;
      pending_head_ = w->next_;
      if (!pending_head_) pending_tail_ = nullptr;
      w->resume_(w);
      return;
    }
    waiter_base* list = try_mark_inactive_or_dequeue_all();
    if (!list) return;

    waiter_base* rev = nullptr;
    while (list) {
      waiter_base* next = list->next_;
      list->next_ = rev;
      rev = list;
      list = next;
    }
    pending_head_ = rev;
    waiter_base* tail = rev;
    while (tail && tail->next_) tail = tail->next_;
    pending_tail_ = tail;

    waiter_base* w = pending_head_;
    pending_head_ = w->next_;
    if (!pending_head_) pending_tail_ = nullptr;
    w->resume_(w);
  }

  auto async_lock() noexcept { return lock_sender{*this}; }

private:
  waiter_base* try_mark_inactive_or_dequeue_all() noexcept {
    uintptr_t old = state_.load(std::memory_order_relaxed);
    do {
      if (old == 0) {
        if (state_.compare_exchange_weak(
              old, INACTIVE, std::memory_order_release,
              std::memory_order_relaxed))
          return nullptr;
      } else {
        waiter_base* list = reinterpret_cast<waiter_base*>(old & ~uintptr_t(1));
        if (state_.compare_exchange_weak(
              old, 0, std::memory_order_acquire,
              std::memory_order_relaxed))
          return list;
      }
    } while (true);
  }

  bool try_enqueue(waiter_base* w) noexcept {
    uintptr_t old = state_.load(std::memory_order_relaxed);
    do {
      uintptr_t new_val;
      if (old & INACTIVE) {
        new_val = 0;                          // inactive → active, no enqueue
      } else {
        w->next_ = reinterpret_cast<waiter_base*>(old & ~uintptr_t(1));
        new_val = reinterpret_cast<uintptr_t>(w);  // push to stack
      }
      if (state_.compare_exchange_weak(
            old, new_val, std::memory_order_acq_rel,
            std::memory_order_relaxed))
        return !(old & INACTIVE);  // true=queued, false=lock acquired
    } while (true);
  }

  class lock_sender {
  public:
    using sender_concept = ex::sender_tag;
    using completion_signatures = ex::completion_signatures<ex::set_value_t()>;

    template <typename R>
    struct op : waiter_base {
      async_mutex& mutex_;
      R rcvr_;
      template <typename R2>
      explicit op(async_mutex& m, R2&& r) noexcept
        : mutex_(m)
        , rcvr_(static_cast<R2&&>(r)) {
        this->resume_ = [](waiter_base* self) noexcept {
          auto& self_op = *static_cast<op*>(self);
          ex::set_value(static_cast<R&&>(self_op.rcvr_));
        };
      }
      op(op&&) = delete;

      void start() & noexcept {
        if (!mutex_.try_enqueue(this)) {
          ex::set_value(static_cast<R&&>(rcvr_));
        }
      }
    };

    template <typename R>
    auto connect(R r) && -> op<R> {
      return op<R>{mutex_, static_cast<R&&>(r)};
    }

    auto get_env() const noexcept { return ex::env<>{}; }

  private:
    friend async_mutex;
    explicit lock_sender(async_mutex& m) noexcept : mutex_(m) {}
    async_mutex& mutex_;
  };
};

async_pass

Core Design

async_pass is a one-to-one synchronous channel (rendezvous). Two roles exchange data through it:

  • Caller: calls async_call(args...), waits for acceptor to receive the parameters
  • Acceptor: calls async_accept(), waits for caller to send parameters

State is encoded in a single std::atomic<uintptr_t>:

  s0: "idle (0)"
  s1: "caller waiting (even, non-zero)"
  s2: "acceptor waiting (odd)"

  s0 --> s1: call() stores caller
  s0 --> s2: accept() stores acceptor
  s1 --> s0: accept() rendezvous
  s2 --> s0: call() rendezvous

Complete Implementation (stdexec)

cpp
template <typename... Args>
class async_pass {
  struct acceptor_base {
    void (*set_value_)(acceptor_base*, Args&&...) noexcept;
    void (*set_error_)(acceptor_base*, std::exception_ptr) noexcept;
    void (*resume_)(acceptor_base*) noexcept;
  };

  struct caller_base {
    void (*invoke_)(caller_base*, acceptor_base&) noexcept;
    void (*resume_)(caller_base*) noexcept;
  };

  static constexpr uintptr_t kAcceptorTag = 1;
  std::atomic<uintptr_t> state_{0};

  static bool is_caller(uintptr_t s) noexcept { return s != 0 && !(s & kAcceptorTag); }
  static bool is_acceptor(uintptr_t s) noexcept { return (s & kAcceptorTag) != 0; }

  static acceptor_base* as_acceptor(uintptr_t s) noexcept {
    return reinterpret_cast<acceptor_base*>(s ^ kAcceptorTag);
  }
  static caller_base* as_caller(uintptr_t s) noexcept {
    return reinterpret_cast<caller_base*>(s);
  }

  acceptor_base* call_or_suspend(caller_base* c) noexcept {
    uintptr_t s = state_.load(std::memory_order_acquire);
    while (true) {
      if (is_acceptor(s)) {
        if (state_.compare_exchange_weak(/*...*/))
          return as_acceptor(s);
      } else if (s == 0) {
        if (state_.compare_exchange_weak(/*...*/))
          return nullptr;
      } else { std::terminate(); }
    }
  }

  caller_base* accept_or_suspend(acceptor_base* a) noexcept { /* symmetric implementation */ }

  class call_sender;
  class accept_sender;

public:
  async_pass() noexcept = default;

  auto async_call(Args... args) noexcept { return call_sender{*this, /*args*/}; }
  auto async_accept() noexcept           { return accept_sender{*this}; }

private:
  class call_sender {
  public:
    using sender_concept = ex::sender_tag;
    using completion_signatures = ex::completion_signatures<ex::set_value_t()>;

    template <typename R>
    struct op : caller_base {
      async_pass& pass_;
      std::tuple<std::decay_t<Args>...> args_;
      R rcvr_;

      template <typename R2>
      explicit op(async_pass& p, auto a, R2&& r) noexcept
        : pass_(p), args_(std::move(a)), rcvr_(static_cast<R2&&>(r)) {
        this->invoke_ = [](caller_base* base, acceptor_base& acc) noexcept {
          auto& self = *static_cast<op*>(base);
          std::apply([&](auto&... as) noexcept {
            (*acc.set_value_)(&acc, std::move(as)...);
          }, self.args_);
        };
        this->resume_ = [](caller_base* base) noexcept {
          auto& self = *static_cast<op*>(base);
          ex::set_value(static_cast<R&&>(self.rcvr_));
        };
      }
      op(op&&) = delete;

      void start() & noexcept {
        if (auto* acc = pass_.call_or_suspend(this)) {
          (*this->invoke_)(this, *acc);
          (*acc->resume_)(acc);
          ex::set_value(static_cast<R&&>(rcvr_));
        }
      }
    };

    template <typename R>
    auto connect(R r) && -> op<R> { return op<R>{pass_, args_, static_cast<R&&>(r)}; }

    auto get_env() const noexcept { return ex::env<>{}; }

  private:
    friend async_pass;
    explicit call_sender(async_pass& p, auto a) noexcept : pass_(p), args_(std::move(a)) {}
    async_pass& pass_;
    std::tuple<std::decay_t<Args>...> args_;
  };

  class accept_sender {
  public:
    using sender_concept = ex::sender_tag;
    using completion_signatures = ex::completion_signatures<
      ex::set_value_t(std::decay_t<Args>...),
      ex::set_error_t(std::exception_ptr)>;

    template <typename R>
    struct op : acceptor_base {
      async_pass& pass_;
      R rcvr_;

      template <typename R2>
      explicit op(async_pass& p, R2&& r) noexcept : pass_(p), rcvr_(static_cast<R2&&>(r)) {
        this->set_value_ = [](acceptor_base* base, Args&&... args) noexcept {
          auto& self = *static_cast<op*>(base);
          ex::set_value(static_cast<R&&>(self.rcvr_), std::forward<Args>(args)...);
        };
        this->set_error_ = [](acceptor_base* base, std::exception_ptr ex) noexcept {
          auto& self = *static_cast<op*>(base);
          ex::set_error(static_cast<R&&>(self.rcvr_), std::move(ex));
        };
        this->resume_ = [](acceptor_base*) noexcept {};
      }
      op(op&&) = delete;

      void start() & noexcept {
        if (auto* caller = pass_.accept_or_suspend(this)) {
          (*caller->invoke_)(caller, *this);
          (*caller->resume_)(caller);
        }
      }
    };

    template <typename R>
    auto connect(R r) && -> op<R> { return op<R>{pass_, static_cast<R&&>(r)}; }

    auto get_env() const noexcept { return ex::env<>{}; }

  private:
    friend async_pass;
    explicit accept_sender(async_pass& p) noexcept : pass_(p) {}
    async_pass& pass_;
  };
};

async_mutex (v2) — With Cancellation Support

Core Design

v2 builds on v1 by adding cancellation support and the Dekker synchronization pattern. Key differences from v1:

  • Lock flag locked_ is separated from the waiter queue, decoupling state management
  • Uses the Dekker pattern (seq_cst fence) to prevent lost wakeups
  • Uses stop_callback + try_remove to implement cancellation while waiting
  • completed_ atomic flag coordinates normal completion and cancellation paths, ensuring exactly one path delivers the completion signal

Complete Implementation (stdexec)

cpp
class async_mutex_v2 {
  struct waiter_base {
    waiter_base* next_ = nullptr;
    void (*resume_)(waiter_base*) noexcept;
  };

  std::mutex mutex_;
  std::deque<waiter_base*> queue_;
  std::atomic<bool> locked_{false};

  class lock_sender;

public:
  bool try_lock() noexcept {
    return !locked_.exchange(true, std::memory_order_acquire);
  }

  void unlock() noexcept {
    process_queue();
  }

  auto async_lock() noexcept { return lock_sender{*this}; }

private:
  void process_queue() noexcept {
    while (true) {
      waiter_base* w = nullptr;
      {
        std::lock_guard lk(mutex_);
        if (!queue_.empty()) {
          w = queue_.front();
          queue_.pop_front();
        }
      }
      if (w) { w->resume_(w); return; }

      locked_.store(false, std::memory_order_release);
      std::atomic_thread_fence(std::memory_order_seq_cst);

      { std::lock_guard lk(mutex_);
        if (queue_.empty()) return; }

      if (locked_.exchange(true, std::memory_order_acq_rel)) return;
    }
  }

  bool try_remove(waiter_base* w) noexcept {
    std::lock_guard lk(mutex_);
    for (auto it = queue_.begin(); it != queue_.end(); ++it)
      if (*it == w) { queue_.erase(it); return true; }
    return false;
  }

  void push_back(waiter_base* w) noexcept {
    std::lock_guard lk(mutex_);
    queue_.push_back(w);
  }

  class lock_sender {
  public:
    using sender_concept = ex::sender_tag;

    template <typename R>
    struct op : waiter_base {
      using stop_token_t = ex::stop_token_of_t<ex::env_of_t<R>>;

      struct stop_handler {
        op* self_;
        void operator()() noexcept { self_->stop(); }
      };
      using cb_t = ex::stop_callback_for_t<stop_token_t, stop_handler>;

      async_mutex_v2& mutex_;
      R rcvr_;
      std::atomic<bool> completed_{false};
      std::optional<cb_t> cb_;
      bool started_ = false;
      bool cancelled_ = false;

      template <typename R2>
      explicit op(async_mutex_v2& m, R2&& r) noexcept
        : mutex_(m), rcvr_(static_cast<R2&&>(r)) {
        this->resume_ = [](waiter_base* self) noexcept {
          auto& s = *static_cast<op*>(self);
          s.cb_.reset();
          if (!s.completed_.exchange(true, std::memory_order_acq_rel)) {
            if (s.cancelled_)
              ex::set_stopped(static_cast<R&&>(s.rcvr_));
            else
              ex::set_value(static_cast<R&&>(s.rcvr_));
          } else {
            s.mutex_.unlock();
          }
        };
      }

      op(op&&) = delete;

      void start() & noexcept {
        auto token = ex::get_stop_token(ex::get_env(rcvr_));
        if (token.stop_requested()) {        // StopsEarly
          ex::set_stopped(static_cast<R&&>(rcvr_));
          return;
        }
        started_ = true;
        if (token.stop_possible())
          cb_.emplace(token, stop_handler{this});
        if (mutex_.try_lock()) {             // Fast path
          cb_.reset();
          if (!completed_.exchange(true, std::memory_order_acq_rel))
            ex::set_value(static_cast<R&&>(rcvr_));
          return;
        }
        mutex_.push_back(this);              // Enqueue
        std::atomic_thread_fence(std::memory_order_seq_cst);  // Dekker
        if (!mutex_.locked_.exchange(true, std::memory_order_acq_rel))
          mutex_.process_queue();
      }

      void stop() noexcept {
        cancelled_ = true;
        if (mutex_.try_remove(this)) {
          cb_.reset();
          if (!completed_.exchange(true, std::memory_order_acq_rel))
            ex::set_stopped(static_cast<R&&>(rcvr_));
        }
      }
    };

    template <typename R>
    auto connect(R r) && -> op<R> {
      return op<R>{mutex_, static_cast<R&&>(r)};
    }

    friend auto tag_invoke(ex::get_completion_signatures_t, auto&&, auto&&) noexcept
      -> ex::completion_signatures<ex::set_value_t(), ex::set_stopped_t()>;

    auto get_env() const noexcept { return ex::env<>{}; }

  private:
    friend async_mutex_v2;
    explicit lock_sender(async_mutex_v2& m) noexcept : mutex_(m) {}
    async_mutex_v2& mutex_;
  };
};

async_manual_reset_event

Core Design

async_manual_reset_event is a manual reset event:

  • set() → transitions to signalled state, wakes all waiters
  • reset() → transitions to not-signalled state
  • ready() → queries current state
  • async_wait() → returns a sender: completes immediately if already signalled, otherwise enqueues and waits

Key differences from async_mutex:

Complete Implementation (stdexec)

cpp
class async_manual_reset_event {
  struct waiter_base {
    waiter_base* next_ = nullptr;
    void (*resume_)(waiter_base*) noexcept;
  };

  std::mutex mutex_;
  std::deque<waiter_base*> waiters_;
  std::atomic<bool> signalled_{false};

  class wait_sender;

public:
  async_manual_reset_event() noexcept = default;
  explicit async_manual_reset_event(bool init) noexcept
    : signalled_(init) {}

  bool ready() const noexcept {
    return signalled_.load(std::memory_order_acquire);
  }

  void set() noexcept {
    std::deque<waiter_base*> to_resume;
    {
      std::lock_guard lk(mutex_);
      signalled_.store(true, std::memory_order_release);
      to_resume.swap(waiters_);
    }
    for (auto* w : to_resume)
      w->resume_(w);
  }

  void reset() noexcept {
    signalled_.store(false, std::memory_order_release);
  }

  auto async_wait() noexcept { return wait_sender{*this}; }

private:
  bool try_enqueue(waiter_base* w) noexcept {
    std::lock_guard lk(mutex_);
    if (signalled_.load(std::memory_order_relaxed)) return false;
    waiters_.push_back(w);
    return true;
  }

  bool try_remove(waiter_base* w) noexcept {
    std::lock_guard lk(mutex_);
    for (auto it = waiters_.begin(); it != waiters_.end(); ++it)
      if (*it == w) { waiters_.erase(it); return true; }
    return false;
  }

  class wait_sender {
  public:
    using sender_concept = ex::sender_tag;

    template <typename R>
    struct op : waiter_base {
      using stop_token_t = ex::stop_token_of_t<ex::env_of_t<R>>;

      struct stop_handler {
        op* self_;
        void operator()() noexcept { self_->stop(); }
      };
      using cb_t = ex::stop_callback_for_t<stop_token_t, stop_handler>;

      async_manual_reset_event& evt_;
      R rcvr_;
      std::atomic<bool> completed_{false};
      std::optional<cb_t> cb_;
      bool cancelled_ = false;

      template <typename R2>
      explicit op(async_manual_reset_event& e, R2&& r) noexcept
        : evt_(e), rcvr_(static_cast<R2&&>(r)) {
        this->resume_ = [](waiter_base* self) noexcept {
          auto& s = *static_cast<op*>(self);
          s.cb_.reset();
          if (!s.completed_.exchange(true, std::memory_order_acq_rel)) {
            if (s.cancelled_)
              ex::set_stopped(static_cast<R&&>(s.rcvr_));
            else
              ex::set_value(static_cast<R&&>(s.rcvr_));
          }
        };
      }

      op(op&&) = delete;

      void start() & noexcept {
        auto token = ex::get_stop_token(ex::get_env(rcvr_));
        if (token.stop_requested()) {        // StopsEarly
          ex::set_stopped(static_cast<R&&>(rcvr_));
          return;
        }
        if (token.stop_possible()) {
          cb_.emplace(token, stop_handler{this});
          if (cancelled_) {                  // Synchronous callback removed us
            if (!completed_.exchange(true, std::memory_order_acq_rel))
              ex::set_stopped(static_cast<R&&>(rcvr_));
            return;
          }
        }
        if (!evt_.try_enqueue(this)) {       // Already signalled
          cb_.reset();
          if (!completed_.exchange(true, std::memory_order_acq_rel))
            ex::set_value(static_cast<R&&>(rcvr_));
        }
      }

      void stop() noexcept {
        if (evt_.try_remove(this)) {
          cb_.reset();
          if (!completed_.exchange(true, std::memory_order_acq_rel))
            ex::set_stopped(static_cast<R&&>(rcvr_));
        } else {
          cancelled_ = true;
        }
      }
    };

    template <typename R>
    auto connect(R r) && -> op<R> {
      return op<R>{evt_, static_cast<R&&>(r)};
    }

    friend auto tag_invoke(ex::get_completion_signatures_t, auto&&, auto&&) noexcept
      -> ex::completion_signatures<ex::set_value_t(), ex::set_stopped_t()>;

    auto get_env() const noexcept { return ex::env<>{}; }

  private:
    friend async_manual_reset_event;
    explicit wait_sender(async_manual_reset_event& e) noexcept : evt_(e) {}
    async_manual_reset_event& evt_;
  };
};

Usage Example

cpp
async_manual_reset_event evt;

// === Signalled → immediate completion ===
evt.set();
ex::sync_wait(evt.async_wait());   // Returns immediately

evt.reset();

// === Not signalled → wait for set() ===
std::atomic<int> count{0};
std::thread t1([&] { ex::sync_wait(evt.async_wait()); count++; });
std::thread t2([&] { ex::sync_wait(evt.async_wait()); count++; });

evt.set();    // Wakes both waiters
t1.join(); t2.join();
// count == 2

// === Cancelling a wait ===
ex::inplace_stop_source src;
bool stopped = false;
auto op = ex::connect(evt.async_wait(), stoppable_receiver{/*token*/});
ex::start(op);
src.request_stop();     // → completed via set_stopped

Summary

Common Patterns

Both primitives share the same design pattern:

  1. Single atomic variable encodes all state (lock + waiters, or caller/acceptor)
  2. CAS loop implements lock-free state transitions
  3. Type-erased function pointers store resumption logic (resume_ / invoke_ / set_value_)
  4. Inline completion on rendezvous synchronously calls set_value, reducing context switches
